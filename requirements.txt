# Use the provided setup instructions below for a clean first install.
#
# QUICK START
# -----------
#   # 1. Create venv
#   python3.12 -m venv .venv && source .venv/bin/activate
#
#   # 2. Install PyTorch with CUDA 12.1 (choose the right CUDA version for your driver)
#   pip install torch==2.3.0 torchaudio==2.3.0 --index-url https://download.pytorch.org/whl/cu121
#
#   # 3. Install everything else
#   pip install -r requirements.txt
#
#   # 4. Install Fish Speech from source (pinned to v1.5 tag)
#   pip install "fish-speech @ git+https://github.com/fishaudio/fish-speech.git@v1.5.0"
#
#   # 5. Download Fish Speech checkpoints to /models/fish-speech-1.5/
#   #    (follow the official fish-speech README for huggingface-hub download)
#
#   # 6. Place a reference speaker WAV at /app/static/reference_speaker.wav
#   #    (required for XTTS; ~10 seconds of clean speech, mono or stereo)
#
#   # 7. Set environment variables (or create a .env file in the project root)
#   export DATABASE_URL="postgresql+asyncpg://tts_user:tts_password@localhost:5432/tts_db"
#
#   # 8. Start the server
#   uvicorn app.main:app --host 0.0.0.0 --port 8000 --workers 1
